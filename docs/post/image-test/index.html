export default function PostTemplate({ data }: { data: Query }) { const post = data.markdownRemark; if (!post) return <div>404</div>; const { frontmatter } = post; const hitsCounter = `dd`; return ( <div> <div> <h1>{frontmatter?.title}</h1> <div dangerouslySetInnerHTML={{ __html: hitsCounter }} /> <div dangerouslySetInnerHTML={{ __html: `${post?.html}` }} /> </div> </div> ); }
